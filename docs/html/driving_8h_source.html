90void dijkstra(Graph<T> * g, const int &source) {
91 // Initialize the vertices
92 for(auto v : g->getVertexSet()) {
93 v->setDist(INF);
94 v->setPath(nullptr);
95 }
96 auto s = g->findVertex(source);
97 s->setDist(0);
98
99 MutablePriorityQueue<Vertex<T>> q;
100 q.insert(s);
101 while( ! q.empty() ) {
102 auto v = q.extractMin();
103 for(auto e : v->getAdj()) {
104 if (!e->getDest()->isVisited()) {
105 auto oldDist = e->getDest()->getDist();
106 if (g->switchwalking) {
107 if (relaxwalking(e)) {
108 if (oldDist == INF) {
109 q.insert(e->getDest());
110 }
111 else {
112 q.decreaseKey(e->getDest());
113 }
114 }
115 } else {
116 if (relaxdriving(e)) {
117 if (oldDist == INF) {
118 q.insert(e->getDest());
119 }
120 else {
121 q.decreaseKey(e->getDest());
122 }
123 }
124 }
125 }
126 }

142template <class T>
143static std::vector<T> getPath(Graph<T> * g, const int &origin, const int &dest) {
144 std::vector<T> res;
145 auto v = g->findVertex(dest);
146 if (v == nullptr || v->getDist() == INF) { // missing or disconnected
147 return res;
148 }
149 if (g->findVertex(origin)->getDist() == INF) { // missing or disconnected
150 return res;
151 }
152 res.push_back(v->getInfo());
153 while(v->getPath() != nullptr) {
154 v = v->getPath()->getOrig();
155 res.push_back(v->getInfo());
156 }
157 reverse(res.begin(), res.end());
158 if(res.empty() || res[0] != origin) {
159 std::cout << "Origin not found!!" << std::endl;
160 }
161 return res;
162}
